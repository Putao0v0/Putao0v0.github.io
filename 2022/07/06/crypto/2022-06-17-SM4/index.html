# SM4

def L(x):
    return x ^ (((x << 2) | (x >> 30)) & 0xffffffff) ^ (((x << 10) | (x >> 22)) & 0xffffffff) ^ \
           (((x << 18) | (x >> 14)) & 0xffffffff) ^ (((x << 24) | (x >> 8)) & 0xffffffff)


def L1(x):
    return x ^ (((x << 13) | (x >> 19)) & 0xffffffff) ^ (((x << 23) | (x >> 9)) & 0xffffffff)


def t(a, b, c, d):
    e = '{:08x}'.format(a ^ b ^ c ^ d)
    a0 = S[int(e[0], 16)][int(e[1], 16)]
    b0 = S[int(e[2], 16)][int(e[3], 16)]
    c0 = S[int(e[4], 16)][int(e[5], 16)]
    d0 = S[int(e[6], 16)][int(e[7], 16)]
    X = int('{:02x}'.format(a0) + '{:02x}'.format(b0) + '{:02x}'.format(c0) + '{:02x}'.format(d0), 16)
    return X

def F(a, b, c, d, s):
    if (s == 0):
        return L(t(a, b, c, d))
    else:
        return L1(t(a, b, c, d))


def key_extend(MK):
    mk0 = int(MK[:8], 16)
    mk1 = int(MK[8:16], 16)
    mk2 = int(MK[16:24], 16)
    mk3 = int(MK[24:32], 16)
    mk0=mk0 ^ FK[0]
    mk1=mk1 ^ FK[1]
    mk2=mk2 ^ FK[2]
    mk3=mk3 ^ FK[3]
    rk = []
    for i in range(32):
        rk.append(mk0 ^ F(mk1, mk2, mk3, CK[i], 1))
        mk0 = mk1
        mk1 = mk2
        mk2 = mk3
        mk3 = rk[i]
    # print(rk)
    return rk


def encrypt(m, key):
    rk = key_extend(key)
    print(rk)
    x0 = int(m[:8], 16)
    x1 = int(m[8:16], 16)
    x2 = int(m[16:24], 16)
    x3 = int(m[24:32], 16)
    for i in range(32):
        temp = x0 ^ F(x1, x2, x3, rk[i], 0)
        x0 = x1
        x1 = x2
        x2 = x3
        x3 = temp
    cipher = '{:08x}'.format(x3) + '{:08x}'.format(x2) + '{:08x}'.format(x1) + '{:08x}'.format(x0)
    return cipher.encode()


def decrypt(c, mk):
    rk = key_extend(mk)
    rk = rk[::-1]
    x0 = int(c[:8], 16)
    x1 = int(c[8:16], 16)
    x2 = int(c[16:24], 16)
    x3 = int(c[24:32], 16)
    for i in range(32):
        temp = x0 ^ F(x1, x2, x3, rk[i], 0)
        x0 = x1
        x1 = x2
        x2 = x3
        x3 = temp
    message = '{:08x}'.format(x3) + '{:08x}'.format(x2) + '{:08x}'.format(x1) + '{:08x}'.format(x0)
    return message.encode()
